class FileMigrator:
    """Migrador de arquivos individuais com checksum"""
    
    def __init__(self, config: MigrationConfig, dry_run: bool = True):
        self.config = config
        self.dry_run = dry_run
        self.migration_log = []
        
    def calculate_checksum(self, filepath: str) -> str:
        """Calcular SHA256 do arquivo para integridade"""
        sha256_hash = hashlib.sha256()
        with open(filepath, "rb") as f:
            for byte_block in iter(lambda: f.read(4096), b""):
                sha256_hash.update(byte_block)
        return sha256_hash.hexdigest()
    
    def should_ignore(self, path: str) -> bool:
        """Verificar se path deve ser ignorado"""
        for pattern in self.config.ignore_patterns:
            if pattern in path:
                return True
        return False
    
    def migrate_file(self, src_path: str, dst_path: str) -> Dict:
        """Migrar arquivo individual com logging"""
        if self.should_ignore(src_path):
            return {"status": "ignored", "reason": "ignore_pattern"}
        
        operation = {
            "timestamp": datetime.now().isoformat(),
            "source": src_path,
            "destination": dst_path,
            "checksum": self.calculate_checksum(src_path),
            "dry_run": self.dry_run
        }
        
        if not self.dry_run:
            try:
                # Criar diretório destino se não existir
                os.makedirs(os.path.dirname(dst_path), exist_ok=True)
                
                # Copiar arquivo preservando metadados
                shutil.copy2(src_path, dst_path)
                
                # Verificar integridade após cópia
                dst_checksum = self.calculate_checksum(dst_path)
                if dst_checksum == operation["checksum"]:
                    operation["status"] = "success"
                    operation["verified"] = True
                else:
                    operation["status"] = "checksum_mismatch"
                    operation["verified"] = False
                    
            except Exception as e:
                operation["status"] = "error"
                operation["error"] = str(e)
        else:
            operation["status"] = "dry_run"
        
        self.migration_log.append(operation)
        return operation

class ObsidianLinkUpdater:
    """Atualizador de links do Obsidian em arquivos Markdown"""
    
    def __init__(self, config: MigrationConfig):
        self.config = config
        self.updated_files = []
        
    def update_file_links(self, filepath: str) -> List[Dict]:
        """Atualizar links em um arquivo Markdown"""
        updates = []
        
        try:
            with open(filepath, 'r', encoding='utf-8') as f:
                content = f.read()
            
            original_content = content
            for pattern, replacement in self.config.link_patterns:
                new_content, count = re.subn(pattern, replacement, content)
                if count > 0:
                    updates.append({
                        "pattern": pattern,
                        "replacement": replacement,
                        "count": count
                    })
                    content = new_content
            
            if content != original_content:
                self.updated_files.append(filepath)
                return updates
                
        except Exception as e:
            logging.error(f"Erro ao processar {filepath}: {e}")
        
        return updates

class LKEMigrationV4:
    """Orquestrador principal da migração LKE v4.0"""
    
    def __init__(self, dry_run: bool = True):
        self.dry_run = dry_run
        self.config = MigrationConfig()
        self.file_migrator = FileMigrator(self.config, dry_run)
        self.link_updater = ObsidianLinkUpdater(self.config)
        self.log_file = f"migration_log_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
        
        # Setup logging
        logging.basicConfig(
            level=logging.INFO,
            format='%(asctime)s - %(levelname)s - %(message)s',
            handlers=[
                logging.FileHandler(f"migration_{datetime.now().strftime('%Y%m%d_%H%M%S')}.log"),
                logging.StreamHandler()
            ]
        )
    
    def scan_directory(self, directory: str) -> List[str]:
        """Scan recursivo de diretório ignorando padrões"""
        files = []
        for root, dirs, filenames in os.walk(directory):
            # Remover diretórios ignorados da lista de walk
            dirs[:] = [d for d in dirs if not self.file_migrator.should_ignore(os.path.join(root, d))]
            
            for filename in filenames:
                filepath = os.path.join(root, filename)
                if not self.file_migrator.should_ignore(filepath):
                    files.append(filepath)
        
        return files
    
    def create_new_structure(self):
        """Criar nova estrutura Johnny.Decimal"""
        new_dirs = [
            "00_SYSTEM/01_SCRIPTS",
            "00_SYSTEM/02_CONFIG", 
            "00_SYSTEM/03_DOCS",
            "00_SYSTEM/04_METADATA",
            "00_SYSTEM/05_ARCHITECTURE",
            
            "11_DOCUMENTS/01_UNIAO_ESTAVEL",
            "11_DOCUMENTS/02_PATRIMONIO",
            "11_DOCUMENTS/03_SOCIETARIO",
            "11_DOCUMENTS/04_SUCESSORIO",
            "11_DOCUMENTS/05_PROVAS",
            
            "12_PRODUCTION/01_PETICOES",
            "12_PRODUCTION/02_PARAGRAFOS",
            "12_PRODUCTION/03_TEMPLATES",
            "12_PRODUCTION/04_DRAFTS",
            
            "21_NOTES/01_CASE_NOTES",
            "21_NOTES/02_INTELLIGENCE",
            "21_NOTES/03_TACTICS",
            
            "99_ARCHIVE/01_BACKUPS",
            "99_ARCHIVE/02_LOGS",
            "99_ARCHIVE/03_CACHE",
            "99_ARCHIVE/04_TRASH",
        ]
        
        for directory in new_dirs:
            if not self.dry_run:
                os.makedirs(directory, exist_ok=True)
                logging.info(f"Criado diretório: {directory}")
            else:
                logging.info(f"[DRY-RUN] Criaria diretório: {directory}")
    
    def migrate_cdd_structure(self):
        """Migrar estrutura CDD para nova taxonomia"""
        logging.info("Iniciando migração da estrutura CDD...")
        
        for src_pattern, dst_pattern in self.config.mapping.items():
            if "06_DOCUMENTS/BRUTOS" in src_pattern:
                # Encontrar todos os arquivos no padrão source
                src_files = self.scan_directory(src_pattern.rstrip('/'))
                
                for src_file in src_files:
                    # Calcular caminho destino
                    relative_path = os.path.relpath(src_file, src_pattern.rstrip('/'))
                    dst_file = os.path.join(dst_pattern.rstrip('/'), relative_path)
                    
                    # Migrar arquivo
                    result = self.file_migrator.migrate_file(src_file, dst_file)
                    logging.info(f"CDD: {src_file} -> {dst_file} [{result['status']}]")
    
    def update_obsidian_links(self):
        """Atualizar links em todas as notas Markdown"""
        logging.info("Atualizando links do Obsidian...")
        
        # Encontrar todas as notas Markdown
        markdown_files = []
        for root, dirs, files in os.walk("."):
            dirs[:] = [d for d in dirs if not self.file_migrator.should_ignore(os.path.join(root, d))]
            for file in files:
                if file.endswith('.md'):
                    markdown_files.append(os.path.join(root, file))
        
        total_updated = 0
        for md_file in markdown_files:
            updates = self.link_updater.update_file_links(md_file)
            if updates:
                total_updated += 1
                if not self.dry_run:
                    # Aplicar atualizações
                    with open(md_file, 'r', encoding='utf-8') as f:
                        content = f.read()
                    
                    for pattern, replacement in self.config.link_patterns:
                        content = re.sub(pattern, replacement, content)
                    
                    with open(md_file, 'w', encoding='utf-8') as f:
                        f.write(content)
                    
                    logging.info(f"Atualizado: {md_file} ({len(updates)} substituições)")
                else:
                    logging.info(f"[DRY-RUN] Atualizaria: {md_file} ({len(updates)} substituições)")
        
        logging.info(f"Total de arquivos atualizados: {total_updated}/{len(markdown_files)}")
    
    def validate_migration(self) -> Dict:
        """Validar integridade da migração"""
        validation = {
            "timestamp": datetime.now().isoformat(),
            "checks": [],
            "passed": 0,
            "failed": 0
        }
        
        # Verificar se nova estrutura existe
        new_dirs = ["00_SYSTEM", "11_DOCUMENTS", "12_PRODUCTION", "21_NOTES", "99_ARCHIVE"]
        for directory in new_dirs:
            check = {"check": f"directory_exists_{directory}", "status": "pending"}
            if os.path.exists(directory):
                check["status"] = "passed"
                validation["passed"] += 1
            else:
                check["status"] = "failed"
                validation["failed"] += 1
            validation["checks"].append(check)
        
        # Verificar integridade de arquivos migrados
        for operation in self.file_migrator.migration_log:
            if operation.get("status") == "success":
                check = {
                    "check": f"file_integrity_{operation['source']}",
                    "status": "pending"
                }
                
                if os.path.exists(operation["destination"]):
                    dst_checksum = self.file_migrator.calculate_checksum(operation["destination"])
                    if dst_checksum == operation["checksum"]:
                        check["status"] = "passed"
                        validation["passed"] += 1
                    else:
                        check["status"] = "checksum_mismatch"
                        validation["failed"] += 1
                else:
                    check["status"] = "file_not_found"
                    validation["failed"] += 1
                
                validation["checks"].append(check)
        
        return validation
    
    def save_log(self):
        """Salvar log completo da migração"""
        log_data = {
            "timestamp": datetime.now().isoformat(),
            "dry_run": self.dry_run,
            "config": {
                "mapping": self.config.mapping,
                "link_patterns": self.config.link_patterns,
                "ignore_patterns": self.config.ignore_patterns
            },
            "operations": self.file_migrator.migration_log,
            "updated_files": self.link_updater.updated_files
        }
        
        with open(self.log_file, 'w', encoding='utf-8') as f:
            json.dump(log_data, f, indent=2, ensure_ascii=False)
        
        logging.info(f"Log salvo em: {self.log_file}")
    
    def run(self):
        """Executar migração completa"""
        logging.info("=" * 60)
        logging.info("INICIANDO MIGRAÇÃO LKE v4.0")
        logging.info(f"Modo: {'DRY-RUN' if self.dry_run else 'EXECUÇÃO'}")
        logging.info("=" * 60)
        
        try:
            # 1. Criar nova estrutura
            self.create_new_structure()
            
            # 2. Migrar estrutura CDD
            self.migrate_cdd_structure()
            
            # 3. Atualizar links Obsidian
            self.update_obsidian_links()
            
            # 4. Validar migração
            if not self.dry_run:
                validation = self.validate_migration()
                logging.info(f"Validação: {validation['passed']} passed, {validation['failed']} failed")
            
            # 5. Salvar log
            self.save_log()
            
            logging.info("=" * 60)
            logging.info("MIGRAÇÃO CONCLUÍDA")
            logging.info("=" * 60)
            
        except Exception as e:
            logging.error(f"Erro durante migração: {e}")
            raise

def main():

- **Probabilidade:** Média
- **Impacto:** Alto (Fabric, MCP, GitHub)
